  "scripts": {
////// ĐỂ DEV
    "watch-sass": "node-sass src/sass/main.scss src/css/style.css -w",                         
    //cái này để compile realtime main.scss về style.css ở folder src
    "devserver":"live-server", // chạy live server
    "start": "npm-run-all --parallel devserver watch:css", // chạy đồng thời cả 2 script khi dùng cờ --parallel

////// ĐỂ NÉN, BUILD 
    "compile:sass": "node-sass src/sass/main.scss src/css/style.comp.css",                     
    // đầu tiên là compile(ko realtime) về style.css 
    "concat:css": "concat -o src/css/style.concat.css src/css/style.comp.css",                
    // tiếp theo là nối các file .css với nhau(trong th có thêm icon.css nữa) và xuất ra file style.concat.css
    "prefix:css":"postcss --use autoprefixer -b \"last 10 versions\" src/css/style.concat.css -o src/css/style.prefix.css",                   
    //cái này thêm các tiền tố hỗ trợ phiên bản web thấp và lạ (-moz,-webkit) và xuất ra file style.prefix.css
    "compress:css":"node-sass src/css/style.prefix.css src/css/style.css --output-style compressed",                                        
    // cuối cùng cái này nén file css lại cho nhẹ hơn
    "build:css":"npm-run-all compile:sass concat:css prefix:css compress:css compress:css",                                                 
    // tổng lại là script chạy đủ 4 cái script trên lần lượt và xuất ra duy nhất 1 file style.css
